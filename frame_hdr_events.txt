If you perform a search:

    metadataStore.analysis.find2(scan_id=2, data=True)

The return value will be formatted like this:
{
    headers: {
        run_hdr_id1: {
            _id: sha1hash
            beamline_id: string
            beamline_config_id : sha1hash
            custom: { free form field }
            end_time: unix datetime
            owner: string
            scan_id: int
            start_time: unix datetime
            status: string (In Progress or Complete)
            tags: list
        },
        run_hdr_id2: {},
        run_hdr_id3: {},
        ...
    },
    beamline_configs: {
        beamline_config_id1: {
            _id: hashed input
            headers: list of run headers specific beamline_config belongs
            beamline_id: string
            custom: { freem form field }
        },
        beamline_config_id2 : {}
        beamline_config_id3 : {}
        ...
    }

    event_descriptors: {
        event_descriptor_id1: {
            id: hashed input
            descriptor_name: string
            event_type_id: integer
            run_header_id: sha1hash
            type_descriptor: string
            data_keys: [
                pv1,
                ...
                pvN
            },
            external_data_keys: {
                ext_pv1,
                ...
                ext_pvM
            },
            aliases: {
                pv1: alias0,
                ...
                pvN: aliasN,
                ext_pv1: ext_alias1,
                ...
                ext_pvM: ext_aliasM,
            }
        },
        event_descriptor_id1: {},
        event_descriptor_id2: {},
        ...
    }
    events: {
        event_id0: {
            _id: hashed input
            data: {
                pv1: python primitive,
                ...
                pvN: python primitive,
                ext_pv1: _id_into_frame_store1,
                ...
                ext_pvM: _id_into_frame_storeM
            }
            time: unix timestamp
            description: string
            event_descriptor_id: sha1hash
            run_header_id: sha1hash
            owner: string
            seq_no: int
        },
        event_id1: {}
        event_id2: {}
        ...
    }
}
